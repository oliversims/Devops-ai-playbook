# Verify boutique app deployment
# Setup: aws eks update-kubeconfig --region us-east-1 --name eks-cluster

kubectl get application boutique -n argocd -o custom-columns=NAME:.metadata.name,SYNC:.status.sync.status,HEALTH:.status.health.status

kubectl get pods -n boutique
kubectl get deployments -n boutique
kubectl get svc -n boutique

kubectl get pods -n boutique -l app=postgres

# Access the app (run each port-forward in a separate terminal)
kubectl port-forward svc/gateway 3001:3001 -n boutique
kubectl port-forward svc/frontend 3000:3000 -n boutique
# Open: http://localhost:3000
# The browser calls the API at http://localhost:3001 — both port-forwards are required
